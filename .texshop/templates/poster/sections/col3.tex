\begin{block}{Mechanism / Interpretation}
\justifying
Translate results into mechanism: "why", supported by evidence.

\begin{tightitemize}
  \item Mechanism point 1 + evidence reference (Fig. X)
  \item Mechanism point 2 + comparison to literature
\end{tightitemize}
\end{block}

\begin{block}{Conclusions \& Outlook}
\justifying
3 conclusions max; 1–2 pragmatic next steps.

\begin{tightitemize}
  \item Conclusion 1
  \item Conclusion 2
  \item Outlook: next step ...
\end{tightitemize}
\end{block}

\begin{block}{References (Tiny)}
\justifying
Poster references are often short. Example:
\begin{tightitemize}
  \item Author et al., \textit{Journal} (Year)
  \item Author et al., \textit{Journal} (Year)
\end{tightitemize}
\end{block}
